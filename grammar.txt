<number> ::= <float> | <int>
<float> ::= <int>.<int>
<int> ::= [0-9]

<newline> ::= \n

<string_internal> ::= [^"] | <string_internal>\"
<string> ::= "<string_internal>"

<literal> ::= <number> | <string> | <identifier>

<identifier> ::= <alpha><identifier> | <identifier><number>

<alpha> ::= [a-z] | [A-Z] | _

<expression> ::= <function_call> | <expression> <operator> <expression> | <literal> | <if_expression> | <while_expression>

<if_expression> ::= if <expression> { <expression_list> } | <if_expression> else <if_expression>

<while_expression> ::= while <expression> { <expression_list> }

<expression_list> ::= <expression> <newline> | <expression_list> <expression> <newline>

<function> ::= function <identifier> (<arg_list>) { <expression_list> }

<function_call> ::= <identifier> (<arg_list>)

<arg_list> ::= <expression> | <expression>, <arg_list>

<operator> ::= + | - | * | / | %

<comment> ::= # <anything> <newline>
